```typescript
type AgentRunMode = "browser-action" | "deep-research";
type AgentRunStatus = "pending" | "running" | "paused" | "waiting_approval" | "completed" | "failed" | "cancelled";
type AgentRunPhase = "init" | "planning" | "executing" | "validating" | "finalizing";

interface AgentRun {
  runId: string;
  mode: AgentRunMode;
  status: AgentRunStatus;
  phase: AgentRunPhase;
  createdAt: number;
  updatedAt: number;
  todoItems: AgentTodoItem[];
  pendingApproval: AgentPendingApproval | null;
  artifactRefs: AgentArtifactRef[];
  latestCheckpointId: string | null;
  terminalOutcome: AgentTerminalOutcome | null;
  metadata: Record<string, unknown>;
}

type AgentRunEvent =
  | RunStartedEvent | RunPhaseChangedEvent | TodoReplacedEvent
  | TodoItemUpsertedEvent | ToolCalledEvent | ToolCompletedEvent
  | ToolFailedEvent | ApprovalRequestedEvent | ApprovalResolvedEvent
  | ArtifactProjectedEvent | EvidenceRecordedEvent | CheckpointCreatedEvent
  | RunCompletedEvent | RunFailedEvent | RunCancelledEvent;
```
